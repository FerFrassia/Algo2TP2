\begin{Interfaz}

  %}

  \textbf{se explica con}: \tadNombre{Secu($\alpha$), Iterador Bidireccional($\alpha$)}.

  \textbf{géneros}: \TipoVariable{lista, itLista($\alpha$)}.

  \Titulo{Operaciones básicas de lista}

  %\InterfazFuncion{NOMBRE}{INPUTS}{TIPO RES}%
  %[ACA VA EL PRE (SI LO HAY)]
  %{ACA VA EL POST}%
  %[$\Theta(COMPLEJIDAD)$]
  %[DESCRIPCION]

  \InterfazFuncion{Pertenece?}{\In{l}{lista}, \In{e}{$\alpha$}}{bool}
  [true]
  {$res \igobs$ está?($l, e$)}
  [$\Complejidad{1}$]
  [Devuelve true o false según si el elemento pertenece o no a la lista]
	
 \end{Interfaz}

\subsection{Algoritmos}

\begin{Algoritmos}
	
	%Algoritmos / Inputs / TSalida / Codigo / Complejidad Final / Justificacion
	\begin{Algoritmo}{Pertenece?}{\In{l}{lista($\alpha$)}, \In{e}{lista}}{bool}
	{
		\State var $itLista$ $\gets$ CrearIt(l) \ComplejidadDer{1}
		\State $res$ $\gets$ false \ComplejidadDer{1}
		\While{HaySiguiente($itLista$) AND $¬res$} \ComplejidadDer{1}
		\If {Siguiente($itLista$) $==$ $e$} \ComplejidadDer{1}
			\State $res$ $\gets$ $true$  \ComplejidadDer{1}
		\EndIf
		\State Avanzar($itLista$) \ComplejidadDer{1}
		\EndWhile
	} 
	{$\Complejidad{1}$}{}
	\end{Algoritmo}

\end{Algoritmos}
